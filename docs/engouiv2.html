Engo UI LIB V2

Create the main UI

local library = loadstring(game:HttpGet("https://raw.githubusercontent.com/joeengo/exploiting/main/EngoUILIB_V2.lua"))()
local main = library:CreateMain(Title, Description, Enum.KeyCode.RightAlt)

Create a tab

local tab = main:CreateTab(name)


Create a textlabel

local label = tab:CreateLabel(text)


Create a button

tab:CreateButton(text, function)


Create a toggle

tab:CreateToggle(text, function)


Create a slider

tab:CreateSlider(name, min, max, function)


Create a textbox

tab:CreateTextbox(name, function)


Create a dropdown

tab:CreateDropdown(name, list, function)


Create a textlist

tab:CreateTextList(name, function)


Create a colorslider

tab:CreateColorSlider(name, function) -- color = Color3.fromHSV(value, 1, 1) to use as a color.


Toggle a toggle

toggle:Toggle()


Update label

label:Update(text)


Add an option to a dropdown

dropdown:CreateOption(option)


Add options to a dropdown

dropdown:CreateOptions(list)


Refresh a dropdown

dropdown:RefreshOptions(list)


Set a colorslider rainbow

colorslider:SetRainbow(true)


Set a colorslider color

colorslider:SetValue(Hue)


Create notification

main:CreateNotification(text, message , function) -- argument is true when they press OK


Create default settings menu

main:CreateSettings()
