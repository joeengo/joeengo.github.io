Engo UI LIB

Create the main UI

local library = loadstring(game:HttpGet("https://raw.githubusercontent.com/joeengo/exploiting/main/EngoUILib"))()
local main = library:CreateUI(name, Enum.KeyCode.RightShift)

Create a tab

local tab = main:CreateTab(name)


Create a textlabel

tab:CreateLabel(text, description)


Create a button

tab:CreateButton(text, description, function() end)


Create a toggle

tab:CreateToggle(text, description, function(callback) end)


Create a slider

tab:CreateSlider(name, min, max, function(value) end)


Create a textbox

tab:CreateTextbox(name, function(input) end)


Create a dropdown

tab:CreateDropdown(name, tableOptions, function(option) end)
